# PyCircuitSim + bsimar (DirectNet LEVEL=73 + BSIM-AR LEVEL=74) runtime deps.
#
# Install with:
#   pip install -r requirements.txt
#
# PyTorch is intentionally NOT pinned here — install the CPU or CUDA build
# that matches your hardware. For the pycircuitsim conda env we currently
# use torch 2.10.0. Example commands:
#   pip install torch                          # CPU
#   pip install torch --index-url https://download.pytorch.org/whl/cu121  # CUDA 12.1
#
# Tsinghua mirror (recommended for users in China — see CLAUDE.md):
#   pip install -i https://pypi.tuna.tsinghua.edu.cn/simple -r requirements.txt

# ── Numerical / plotting core ───────────────────────────────────────────────
# numpy 2.x is required (installed env uses 2.4.2); 1.26 is the last 1.x line
# that is API-compatible with our code and is a safe lower bound.
numpy>=1.26
# scipy.sparse (lil_matrix, csr, spsolve) — the sparse MNA solver.
# scipy.ndimage / scipy.signal.windows — LDS kernel smoothing in bsimar.losses.
scipy>=1.11
matplotlib>=3.7

# ── scikit-learn ────────────────────────────────────────────────────────────
# sklearn.preprocessing.KBinsDiscretizer — LDS weight binning in
#   bsimar.losses.bni_mae.compute_lds_weights_per_target
# sklearn.metrics.r2_score — scatter plots in bsimar.eval.visualization
scikit-learn>=1.3

# ── pybind11 ────────────────────────────────────────────────────────────────
# Required by the PyCMG submodule (used by pycircuitsim.models.mosfet_cmg
# and the NN data generator at external_compact_models/PyCMG/scripts/).
pybind11>=2.11

# ── Testing ─────────────────────────────────────────────────────────────────
pytest>=7.0

# ── PyTorch (install separately) ─────────────────────────────────────────────
